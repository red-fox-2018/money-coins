STORE 'coin' with array number value
STORE 'result' with array no value
STORE 'i' with 0
FOR 'i' equal to 0, 'i' less than 'coint' length, ADD 'i' by 1
    WHILE 'num' less than 'coin' index 'i'

      ADD 'result' by 'coin' index 'i'
RETURN result
